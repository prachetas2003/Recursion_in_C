#include <stdio.h>

// Recursive function to calculate the factorial of n
int fact(int n)
{
    // Base case: if n is 0, the factorial of 0 is 1
    if (n == 0)
        return 1;

    // Recursive case: multiply n by the factorial of (n-1)
    return fact(n-1) * n;
}

// Iterative function to calculate the factorial of n
int Ifact(int n)
{
    int s = 1;  // Initialize the result variable s to 1
    int i;      // Declare loop variable i

    // Loop to multiply s by every number from 1 to n
    for (i = 1; i <= n; i++)
        s = s * i;

    return s;
}

int main()
{
    // Compute the factorial of 5 using the recursive function
    int r = fact(5);
    printf("Recursive: %d\n", r);

    // Compute the factorial of 5 using the iterative function
    int iter_r = Ifact(5);
    printf("Iterative: %d\n", iter_r);

    return 0;
}
