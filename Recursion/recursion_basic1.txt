#include <stdio.h>
#include <stdlib.h>

void func(int x){  // here, 
    if(x>0){
        func(x-1); // this call is executed first and only after this finshes, will
        printf("%d",x); // the complier come to this line.
    }            // so, the order of function calls is : func(3){from main} -> func(2) -> func(1) -> func(0)
    return; // since func(0) does nothing, we will come back to func(1) from where we left and print 1
} // and like that, we will complete all the remaining calls , hence giving the output of -> 123

int main(){
    int a = 3;
    func(a);
    return 0;
}
