#include <stdio.h>
#include <stdlib.h>

struct rectangle{
    int length;
    int breadth;
};

void changelen(struct rectangle * R1,int new){ // here r1 is a pointer pointing to the address of R 
    R1->length = new;
    return;
}


int main()
{
    struct rectangle R;
    R.length = 15;
    R.breadth = 30;
    changelen(&R,40);
    
    
    printf("%d\n",R.length);
    printf("%d\n",R.breadth);
    

    return 0;
}
